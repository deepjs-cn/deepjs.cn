v-model 指令，实际上就是

import Vue from 'vue'

const VInput = {
  props: ['value'],
  template: `
    <div>
      <input type="text" @input="handleInput" :value="value">
    </div>
  `,
  methods: {
    handleInput (e) {
      this.$emit('input', e.target.value)
    }
  }
}

new Vue({
  components: {
    VInput,
  },
  el: '#root',
  template: `
    <div>
      <v-input :value="value" @input="value = arguments[0]"></v-input>
    </div>
  `,
  data() {
    return {
      value: '',
    }
  },
})
